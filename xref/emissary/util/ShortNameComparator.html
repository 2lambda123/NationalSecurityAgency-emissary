3   import emissary.core.Family;
4   import emissary.core.IBaseDataObject;
5   
6   import org.apache.commons.lang3.Validate;
7   
8   import java.io.Serializable;
9   import java.util.Comparator;
10  
11  /**
12   * Allow a Collection or Array of IBaseDataObject to be sorted by shortName such that all attachments come in order and
13   * all parents are immediately followed by their children and all siblings are in numerical (i.e. not string) order.
14   */
15  public class ShortNameComparator implements Comparator<IBaseDataObject>, Serializable {
16  
17      // Serializable
18      static final long serialVersionUID = -7621558910791975422L;
19  
20      @Override
21      public int compare(IBaseDataObject obj1, IBaseDataObject obj2) {
22          Validate.isTrue(obj1 != null, "Required: obj1 != null");
23          Validate.isTrue(obj2 != null, "Required: obj2 != true");
24  
25          String s1 = obj1.shortName();
26          String s2 = obj2.shortName();
27          int index1 = s1.indexOf(Family.SEP);
28          int index2 = s2.indexOf(Family.SEP);
29  
30          // Loop through each level in obj1's shortName.
31          while (index1 != -1) {
32              // If obj1's shortName has more levels than obj2's shortName then obj1 > obj2.
33              if (index2 == -1) {
34                  return 1;
35              }
36  
37              // Get start character of level.
38              index1 += Family.SEP.length();
39              index2 += Family.SEP.length();
40  
41              // Get character length of level.
42              final int nextIndex1 = s1.indexOf(Family.SEP, index1);
43              final int nextIndex2 = s2.indexOf(Family.SEP, index2);
44              final int length1 = ((nextIndex1 < 0) ? s1.length() : nextIndex1) - index1;
45              final int length2 = ((nextIndex2 < 0) ? s2.length() : nextIndex2) - index2;
46  
47              // If the obj1's level and obj2's level are the same, then go to the next level.
48              if (length1 == length2 && s1.regionMatches(index1, s2, index2, length1)) {
49                  index1 = nextIndex1;
50                  index2 = nextIndex2;
51  
52                  continue;
53              }
54  
55              // Otherwise, try comparing the levels as integers.
56              try {
57                  int int1 = Integer.parseInt(s1, index1, index1 + length1, 10);
58                  int int2 = Integer.parseInt(s2, index2, index2 + length2, 10);
59  
60                  return int1 - int2;
61              } catch (NumberFormatException e) {
62                  // Otherwise, compare the levels as strings.
63                  final String substring1 = s1.substring(index1, index1 + length1);
64                  final String substring2 = s2.substring(index2, index2 + length2);
65  
66                  return substring1.compareTo(substring2);
67              }
68          }
69  
70          // If obj1's shortName has fewer levels than obj2's shortName then obj1 < obj2.
71          if (index2 != -1) {
72              return -1;
73          }
74  
75          // Otherwise, obj1 and obj2 compare as equivalent
76          return 0;
77      }
78  }
