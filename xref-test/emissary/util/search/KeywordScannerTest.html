122     @Test
123     void testIndexOf() {
124         assertEquals(33, this.ks.indexOf("road".getBytes()));
125         assertEquals(22, this.ks.indexOf("Emergency".getBytes(), 0));
126         assertEquals(-1, this.ks.indexOf("Emergency".getBytes(), 40));
127         assertEquals(12, this.ks.indexOf("st".getBytes(), 0, 30));
128         assertEquals(12, this.ks.indexOf("st".getBytes(), 0, DATA.length));
129         assertEquals(12, this.ks.indexOf("st".getBytes(), 12, DATA.length));
130         assertEquals(39, this.ks.indexOf("st".getBytes(), 30, 41));
131         assertEquals(39, this.ks.indexOf("st".getBytes(), 39, DATA.length));
132         // no matches in this range
133         assertEquals(-1, this.ks.indexOf("st".getBytes(), 30, 40));
134         assertEquals(22, this.ks.indexOf("E".getBytes(), 0, DATA.length));
135         // start offset < 0
136         assertEquals(22, this.ks.indexOf("E".getBytes(), -1, DATA.length));
137         // stop index > data length
138         assertEquals(-1, this.ks.indexOf("st".getBytes(), 0, DATA.length + 1));
139         // start offset == data length
140         assertEquals(-1, this.ks.indexOf("st".getBytes(), DATA.length, DATA.length));
141         // start offset > data length
142         assertEquals(-1, this.ks.indexOf("st".getBytes(), DATA.length + 1, DATA.length));
143         // pattern is null
144         assertEquals(-1, this.ks.indexOf(null, 0, DATA.length));
145         // stop index is exclusive
146         assertEquals(-1, this.ks.indexOf("HI".getBytes(), 0, 2));
147         assertEquals(1, this.ks.indexOf("HI".getBytes(), 0, 3));
148         // negative stop index results in failure to match
149         assertEquals(-1, this.ks.indexOf("HI".getBytes(), 0, -1));
150         assertEquals(-1, this.ks.indexOf("HI".getBytes(), 0, -5));
151         // stop < start
152         assertEquals(-1, this.ks.indexOf("i".getBytes(), 5, 1));
153         // stop == start
154         assertEquals(-1, this.ks.indexOf("i".getBytes(), 5, 5));
155         // pattern is longer than stop - start
156         assertEquals(-1, this.ks.indexOf("Emergency".getBytes(), 0, 5));
157         // pattern is longer than the data
158         assertEquals(-1, this.ks.indexOf(new byte[75], 0, DATA.length));
159 
160         // try using an array of negative byte values and a negative byte pattern
161         this.ks = new KeywordScanner(new byte[] {-1, -1, -1, -3, -5, -7});
162         assertEquals(2, this.ks.indexOf(new byte[] {-1, -3, -5}));
163     }

165     @Test
166     void testFindNext() {
167         // no pattern set
168         assertEquals(-1, this.ks.findNext());
169         assertEquals(-1, this.ks.findNext(20));
170         // sets the pattern and returns first match
171         assertEquals(12, this.ks.indexOf("st".getBytes()));
172         assertEquals(39, this.ks.findNext());
173         assertEquals(47, this.ks.findNext());
174         assertEquals(-1, this.ks.findNext());
175         // does not loop back around
176         assertEquals(-1, this.ks.findNext());
177         // sets a new pattern and returns first match
178         assertEquals(8, this.ks.indexOf("a".getBytes()));
179         // negative stop
180         assertEquals(-1, this.ks.findNext(-1));
181         // stop > data length
182         assertEquals(-1, this.ks.findNext(75));
183         // stop < previous position
184         assertEquals(-1, this.ks.findNext(5));
185         // stop == data length
186         assertEquals(-1, this.ks.findNext(DATA.length));
187         assertEquals(35, this.ks.findNext(DATA.length - 1));
188         // changing the pattern, we should have reset the position
189         assertEquals(6, this.ks.indexOf("s".getBytes()));
190         // stop is exclusive
191         assertEquals(12, this.ks.findNext(13));
192         // use a pattern that won't hit
193         assertEquals(-1, this.ks.indexOf("X".getBytes()));
194         // new pattern must be set in failure or we would get an index off the old one
195         assertEquals(-1, this.ks.findNext());
196         // change the pattern back
197         assertEquals(6, this.ks.indexOf("s".getBytes()));
198         // stop is short of the next position (12)
199         assertEquals(-1, this.ks.findNext(12));
200         // we can't continue from last successful index returned, a bad stop ends it
201         assertEquals(-1, this.ks.findNext());
202     }
