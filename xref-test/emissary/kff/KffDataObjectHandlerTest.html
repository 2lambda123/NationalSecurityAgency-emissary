134     @Test
135     void testHashMethodCalledTwice() {
136         // don't truncate known data or the second call will be made with an empty payload
137         kff = new KffDataObjectHandler(KffDataObjectHandler.KEEP_KNOWN_DATA, true, true);
138         payload.setParameter(KffDataObjectHandler.KFF_PARAM_KNOWN_FILTER_NAME, "test.filter");
139         kff.hash(payload);
140 
141         assertNull(payload.getStringParameter(KffDataObjectHandler.MD5_ORIGINAL),
142                 "MD5_ORIGINAL should only be populated if hashing more than once");
143 
144         // hash again, to see the effect on the hash-related params.
145         // none of the parameters should have a duplicated value
146 
147         kff.hash(payload);
148         assertEquals("test.filter", payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_KNOWN_FILTER_NAME));
149         assertTrue(KffDataObjectHandler.hashPresent(payload));
150         assertEquals(DATA_MD5, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_MD5));
151         assertEquals(DATA_CRC32, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_CRC32));
152         assertEquals(DATA_SSDEEP, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SSDEEP));
153         assertEquals(DATA_SHA1, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA1));
154         assertEquals(DATA_SHA256, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA256));
155         assertEquals(KffDataObjectHandler.KFF_DUPE_CURRENT_FORM, payload.getFileType());
156         assertNull(payload.getStringParameter(KffDataObjectHandler.MD5_ORIGINAL),
157                 "MD5_ORIGINAL should not be populated if hash called more than once but data hasn't changed");
158     }

160     @Test
161     void testHashMethodAgainAfterModifyingData() {
162         // don't truncate known data or the second call will be made with an empty payload
163         kff = new KffDataObjectHandler(KffDataObjectHandler.KEEP_KNOWN_DATA, true, true);
164         payload.setParameter(KffDataObjectHandler.KFF_PARAM_KNOWN_FILTER_NAME, "test.filter");
165         kff.hash(payload);
166 
167         assertEquals("test.filter", payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_KNOWN_FILTER_NAME));
168         assertTrue(KffDataObjectHandler.hashPresent(payload));
169         assertEquals(DATA_MD5, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_MD5));
170         assertEquals(DATA_CRC32, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_CRC32));
171         assertEquals(DATA_SSDEEP, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SSDEEP));
172         assertEquals(DATA_SHA1, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA1));
173         assertEquals(DATA_SHA256, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA256));
174         assertNull(payload.getStringParameter(KffDataObjectHandler.MD5_ORIGINAL),
175                 "MD5_ORIGINAL should only be populated if hashing more than once and data has changed");
176 
177         payload.setData("This is a changed data".getBytes());
178         // hash again, to see the effect on the hash-related params.
179         // none of the parameters should have a duplicated value
180 
181         kff.hash(payload);
182         assertEquals("test.filter", payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_KNOWN_FILTER_NAME));
183         assertTrue(KffDataObjectHandler.hashPresent(payload));
184         assertNotNull(payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_MD5));
185         assertNotNull(payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_CRC32));
186         assertNotNull(payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SSDEEP));
187         assertNotNull(payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA1));
188         assertNotNull(payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA256));
189         assertNotEquals(DATA_MD5, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_MD5));
190         assertNotEquals(DATA_CRC32, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_CRC32));
191         assertNotEquals(DATA_SSDEEP, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SSDEEP));
192         assertNotEquals(DATA_SHA1, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA1));
193         assertNotEquals(DATA_SHA256, payload.getStringParameter(KffDataObjectHandler.KFF_PARAM_SHA256));
194         assertEquals(KffDataObjectHandler.KFF_DUPE_CURRENT_FORM, payload.getFileType());
195 
196         // make sure we've correctly populated MD5_ORIGINAL
197         assertNotNull(payload.getStringParameter(KffDataObjectHandler.MD5_ORIGINAL),
198                 "MD5_ORIGINAL should be populated if hash called more than once");
199         assertEquals(DATA_MD5, payload.getStringParameter(KffDataObjectHandler.MD5_ORIGINAL));
200     }
