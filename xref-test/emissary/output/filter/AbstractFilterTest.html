35      AbstractFilter getDenylistFilter(final List<String> denylist) {
36          Configurator config = new ServiceConfigGuide();
37          config.addEntry("OUTPUT_SPEC_FOO", "SPEC");
38          config.addEntry("OUTPUT_TYPE", "TYPE");
39          denylist.forEach(entry -> config.addEntry("DENYLIST", entry));
40          AbstractFilter f = getAbstractFilterInstance();
41          f.initialize(new ServiceConfigGuide(), "FOO", config);
42          return f;
43      }

58          @Test
59          void testDefaultDenylistFiletypeFormat() {
60              String pattern = f.getDenylistFiletypeFormat();
61              assertEquals("^[a-zA-Z0-9_\\-]+$", pattern);
62  
63              assertFalse(f.matchesDenylistFiletypeFormatPattern(""),
64                      String.format("Unexpected match [empty string] for Pattern %s", pattern));
65  
66              String formatString = "Fi1e%sTyp3";
67  
68              String noMatchChars = " !@#$%^&*()+=\\/.,";
69              for (int i = 0; i < noMatchChars.length(); i++) {
70                  char noMatchChar = noMatchChars.charAt(i);
71                  String noMatch = String.format(formatString, noMatchChar);
72                  assertFalse(f.matchesDenylistFiletypeFormatPattern(noMatch),
73                          String.format("Unexpected match %s for Pattern %s", noMatch, pattern));
74              }
75  
76              String nullInsert = String.format(formatString, "");
77              assertTrue(f.matchesDenylistFiletypeFormatPattern(nullInsert),
78                      String.format("Expected match %s for Pattern %s", nullInsert, pattern));
79  
80              String matchChars = "-_";
81              for (int i = 0; i < matchChars.length(); i++) {
82                  char matchChar = matchChars.charAt(i);
83                  String match = String.format(formatString, matchChar);
84                  assertTrue(f.matchesDenylistFiletypeFormatPattern(match),
85                          String.format("Expected match %s for Pattern %s", match, pattern));
86              }
87          }

127     @Test
128     void testIncorrectConfigs() {
129         List<String> invalidEntries = Arrays.asList(
130                 "type*.view", "*type.view", "ty*pe.view", "*.view", ".view",
131                 "type*.*view", "*type.*view", "ty*pe.*view", "*.*view", ".*view", "*view",
132                 "type*.vi*ew", "*type.vi*ew", "ty*pe.vi*ew", "*.vi*ew", ".vi*ew", "vi*ew",
133                 "type*.view*", "*type.view*", "ty*pe.view*", "*.view*", ".view*",
134                 "type.", "*", "type.*", ".", "type.view.", ".type.view", ".type.view.",
135                 "type..view", "type.view.view.view", "type.view*.view*", "type.view**", "view**");
136         for (String entry : invalidEntries) {
137             final Configurator config = new ServiceConfigGuide();
138             config.addEntry("DENYLIST", entry);
139             AbstractFilter f = getAbstractFilterInstance();
140             EmissaryRuntimeException e = assertThrows(
141                     EmissaryRuntimeException.class,
142                     () -> f.initialize(new ServiceConfigGuide(), "FOO", config));
143             assertTrue(e.getMessage().contains(String.format("Invalid filter configuration: `DENYLIST = \"%s\"`", entry)));
144         }
145     }

147     @Test
148     void testDenyListContains() {
149         AbstractFilter f = getDenylistFilter(Arrays.asList(
150                 "JSON_ML", "JSON_LANG_*",
151                 "JSON.GeoJSON", "JSON.JSON_1_*"));
152 
153         String filetype1 = "JSON";
154         for (String accepted : Arrays.asList(
155                 "PrimaryView", "JSON_PRETTY", "JSON_LANG", "Geo", "JSON_1", "JSON_2_10", "JSON_2_10.1")) {
156             assertFalse(f.denyListContains(filetype1, accepted), accepted + " should not be in denyList");
157         }
158

161                 "GeoJSON", "JSON_1_", "JSON_1_0", "JSON_1_2", "JSON_1_2.1")) {
162             assertTrue(f.denyListContains(filetype1, denied), denied + " should be in denyList");
163         }
164 
165         String filetype2 = "XML";

168                 "JSON_1_", "JSON_1", "JSON_1_0", "JSON_1_2", "JSON_1_2.1", "JSON_2_10", "JSON_2_10.1")) {

201                 "*.AlternateView", "NONE.Language", "*.Language", "JSON")) {
202             assertTrue(checkTypes.contains(checked), checked + " should be a checked type");
203         }
204 
205         for (String notChecked : Arrays.asList(
