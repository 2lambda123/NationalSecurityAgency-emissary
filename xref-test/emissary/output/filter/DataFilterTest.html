37  
38          IDropOffFilter f = new DataFilter();

40  
41          IBaseDataObject payload = DataObjectFactory.getInstance();
42          payload.setData("This is the data".getBytes());
43          payload.setFileType("FTYPE");
44          payload.setFilename("/this/is/a/testfile");
45  
46          Map<String, Object> params = new HashMap<>();
47  
48          int status = f.filter(payload, params);
49  
50          File expected = new File("/tmp/testfile.FTYPE");
51          assertEquals(IDropOffFilter.STATUS_SUCCESS, status, "Status of filter should be success");
52          assertTrue(expected.exists(), "Output File should exist");
53  
54          String output = new String(Executrix.readDataFromFile("/tmp/testfile.FTYPE"));
55          assertEquals(new String(payload.data()), output, "Output must be the payload and nothing else");
